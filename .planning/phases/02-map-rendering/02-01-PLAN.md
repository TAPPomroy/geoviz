Private Sub AppendLine(ByRef sb As String, ByVal line As String):
  Appends `line & vbLf` to `sb`. No return value; mutates sb by reference. This is the sole string-building primitive used throughout BuildMapHtml.

Public Sub WriteUtf8File(filePath As String, content As String):
  Creates an ADODB.Stream object via CreateObject("ADODB.Stream").
  Sets .Type = 2 (adTypeText), .Charset = "utf-8", then calls .Open, .WriteText content, .SaveToFile filePath 2 (adSaveCreateOverWrite), .Close.
  Sets the object to Nothing after Close.
  Must NOT use Scripting.FileSystemObject — FSO defaults to ANSI and corrupts non-ASCII characters.
  No error handler needed in this sub; the caller (GenerateMap in mod_Macros) handles errors.
