      function buildLegend() {
        if (!legendDiv) return;
        var html = '<strong>' + currentField + '</strong><br>';
        fieldNames.forEach(function(v) {
          var color = fieldColorMap[v] || OTHER_COLOR;
          var hidden = hiddenValues[v] ? true : false;
          var opacity = hidden ? '0.4' : '1';
          var strike = hidden ? 'text-decoration:line-through;' : '';
          html += '<span style="display:inline-block;width:12px;height:12px;background:' + color
               + ';margin-right:6px;border-radius:2px;vertical-align:middle;opacity:' + opacity + '"></span>'
               + '<span style="cursor:pointer;' + strike + '" onclick="toggleValue(\'' + v.replace(/'/g, "\\'") + '\')">'
               + v + '</span><br>';
        });
        if (fieldNames.length > ISP_COLORS.length) {
          html += '<span style="display:inline-block;width:12px;height:12px;background:#999999;margin-right:6px;border-radius:2px;vertical-align:middle"></span>Other<br>';
        }
        legendDiv.innerHTML = html;
      }

    IMPORTANT for VBA string generation: The onclick attribute uses single quotes around the JS string argument. The value v may contain single quotes — apply .replace(/'/g, "\\'") to escape them in the HTML attribute. In VBA AppendLine, this JS single-quote escaping is straightforward since the VBA string itself uses double quotes at the outer boundary.

      function toggleValue(val) {
        if (hiddenValues[val]) {
          delete hiddenValues[val];
          markers.forEach(function(m, i) {
            if (String(companies[i][currentField]) === val) {

            }
          });
        }
        buildLegend();
      }

    applyFilter(field) — changes currentField, resets hidden state, recolors all markers, rebuilds legend, clears any active selection (per open question resolution in RESEARCH.md):
      function applyFilter(field) {
        clearSelection();
        currentField = field;
        hiddenValues = {};
        buildColorMap(field);
        markers.forEach(function(m, i) {
          var v = companies[i][field] !== undefined ? String(companies[i][field]) : '';
          m.setStyle({ fillColor: getFieldColor(v) });
          // Ensure marker is visible (switching filter resets all to shown, D-13)
          clusterGroup.addLayer(m);
        });
        buildLegend();
      }

    NOTE: clusterGroup.addLayer(m) inside applyFilter is idempotent — calling it on a marker already in the group is a no-op in Leaflet.markercluster. This is the correct way to ensure all markers are visible after a filter switch (D-13: switching filter column resets all visibility to shown).

    Filter dropdown Leaflet control. Position: topright (above the legend, which is bottomright — no conflict). Add a new L.control for the dropdown:
      var filterControl = L.control({position:'topright'});
      filterControl.onAdd = function(map) {
        var div = L.DomUtil.create('div', 'filter-control');
        div.style.cssText = 'background:white;padding:8px 12px;border-radius:4px;font-size:13px;box-shadow:0 1px 5px rgba(0,0,0,.3)';
        var SYSTEM_FIELDS = ['Lat','Lon','GeocodedAt'];
        var filterFields = [];
        if (companies.length > 0) {
          Object.keys(companies[0]).forEach(function(k) {
            if (SYSTEM_FIELDS.indexOf(k) === -1) { filterFields.push(k); }
          });
        }
        var opts = filterFields.map(function(f) {
          return '<option value="' + f + '"' + (f === 'ISP' ? ' selected' : '') + '>' + f + '</option>';
        }).join('');
        div.innerHTML = '<label style="display:block;margin-bottom:4px"><strong>Color by</strong></label>'
                      + '<select id="fieldSelect" style="font-size:13px;width:100%">' + opts + '</select>';
        L.DomEvent.disableClickPropagation(div);
        L.DomEvent.disableScrollPropagation(div);
        return div;
      };
      filterControl.addTo(map);

    Bind the dropdown change event after addTo(map):
